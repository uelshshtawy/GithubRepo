#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
#include  <bits/stdc++.h>
using namespace std;


int main() {
   
  int rows;
  int cols;
  cin >> rows;
  cin >> cols;
  int just=0;
  int vill=0;
  
  int arr1[rows][cols];
  for (int i = 0; i < rows; ++i) {
    for (int j = 0; j < cols; ++j) {
      cin >> arr1[i][j];
    }
  }

  int arr2[rows][cols];
  for (int i = 0; i < rows; ++i) {
    for (int j = 0; j < cols; ++j) {
      cin >> arr2[i][j];
    }
  }

  for (int i = 0; i < rows; ++i) {
    for (int j = 0; j < cols; ++j) {
      if(arr1[i][j] > arr2[i][j]){
        just++;

      }else if(arr1[i][j] < arr2[i][j]){
       vill++;

    }else{
    vill++;
    just++;}
    }
  }
    
  if(just>vill){
    cout << "Justice League";
  }else if(just<vill){
    cout << "Villains";
  }else{
    cout << "Tie";
  }
return 0;
    
}
